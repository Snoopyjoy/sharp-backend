    /**

    }

    /**
     * @description 检查邮件地址是否规范
     * @param {string}str 邮件地址
     * @return {boolean}
     */
    static checkEmailFormat( str ){
        if ( !str || typeof str != 'string') return false;
        const re = /^(\w-*\.*)+@(\w-?)+(\.\w{2,})+$/;
        return re.test(str);
    }
}

module.exports = Utils;
